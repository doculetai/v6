## Test Helpers

```typescript
// tests/helpers/auth.ts
export async function loginAs(page: Page, role: 'student' | 'sponsor' | 'admin' | 'agent' | 'partner' | 'university') {
  const credentials = { ... }; // from seed data
  await page.goto('/login');
  await page.fill('[name="email"]', credentials[role].email);
  await page.fill('[name="password"]', credentials[role].password);
  await page.click('button[type="submit"]');
  await page.waitForURL(`/dashboard/${role}`);
}
```
